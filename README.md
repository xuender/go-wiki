# GoWiki
```
# 创建项目
ng new go-wiki --routing
# 增加 angular material 及动画依赖
npm install --save @angular/material @angular/cdk @angular/animations
# 增加angular material
ng add @angular/material
# 增加组件开发套装(component Dev Kit)
ng add @angular/cdk

# 创建导航页
ng generate @angular/material:nav home
# 编辑页
ng generate @angular/material:address-form editor
# 启动服务
ng serve
```
